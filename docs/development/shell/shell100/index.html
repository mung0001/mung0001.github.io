    #!/bin/bash


Solution(){
        Solution=()
        i=0
        while [ $i -le 3 ];
        do
                Solution[$i]=$((RANDOM% 9+1))

                if [ $i -eq 3 ]; then
                        if [ ${Solution[3]} -eq ${Solution[0]} -o ${Solution[3]} -eq ${Solution[1]} -o ${Solution[3]} -eq ${Solution[2]} ]; then
                                Solution[$i]=$((RANDOM% 9+1))
                        else
                                ((i++))
                        fi
                fi

                if [ $i -eq 2 ]; then
                        if [ ${Solution[2]} -eq ${Solution[0]} -o ${Solution[2]} -eq ${Solution[1]} ]; then
                                Solution[$i]=$((RANDOM% 9+1))
                        else
                                ((i++))
                        fi
                fi

                if [ $i -eq 1 ]; then
                        if [ ${Solution[0]} -eq ${Solution[1]} ]; then
                                Solution[$i]=$((RANDOM% 9+1))
                        else
                                ((i++))
                        fi
                fi

                if [ $i -eq 0 ]; then
                        ((i++))
                fi

        done
        echo ${Solution[@]}
}

Count=0

Score(){

        Out=0
        Ball=0
        Strike=0

        if [ $num1 -eq ${Solution[0]} ]; then
                ((Strike++))
        elif [ $num1 -eq ${Solution[1]} -o $num1 -eq ${Solution[2]} -o $num1 -eq ${Solution[3]} ]; then
                ((Ball++))

                ((Out++))
        fi

                ((Ball++))

    #!/bin/bash

for i in {1..32};
do
        echo "- - -" > "/sys/class/scsi_host/host$i/scan"
done


Mount(){ 
        if [ ! -d ~/Mountlist ]; then
                mkdir ~/Mountlist
                echo `ls /dev/* > ~/Mountlist/Mountlist.txt`
                Mount
        elif [ -d ~/Mountlist -a ! -f ~/Mountlist/Mountlist.txt ]; then
                echo `ls /dev/* > ~/Mountlist/Mountlist.txt`
                Mount
        elif [ -d ~/Mountlist -a -f ~/Mountlist/Mountlist.txt ]; then
                echo `ls /dev/* > ~/Mountlist/Mountlist2.txt`
        fi


        diff ~/Mountlist/Mountlist.txt ~/Mountlist/Mountlist2.txt | grep /dev/ > ~/Mountlist/Mount.txt
        sed -e 's/> //g' ~/Mountlist/Mount.txt > ~/Mountlist/Mount2.txt
        rm -rf ~/Mountlist/Mount.txt
        Mount=`cat ~/Mountlist/Mount2.txt`

        fdisk
}

fdisk(){
        echo 



        Num=1
        for i in $Mount;
        do
        cat ~/Mountlist/Partition.txt | grep Create > ~/Mountlist/Create.txt
        Create_Partition_Number=`sed -e 's/Create Partition Number 0~4 ://g' ~/Mountlist/Create.txt`
        if [ -z "$Create_Partition_Number" -o "Create_Partition_Number" == 0 ]; then
                Create_Partition_Number=1
        fi

        cat ~/Mountlist/Partition.txt | grep type > ~/Mountlist/Create.txt
        Partition_type=`sed -e 's/Partition type ://g' ~/Mountlist/Create.txt`
        if [ -z "$Partition_type" ]; then
                Partition_type=p
        elif [ "$Partition_type" == "p" -o "$Partition_type" == "e" ]; then
                echo "잘못된 설정을 기입하여 기본설정으로 변경합니다."
                Partition_type=p
        fi
        
        cat ~/Mountlist/Partition.txt | grep First > ~/Mountlist/Create.txt
        First_sector=`sed -e 's/First sector ://g' ~/Mountlist/Create.txt`
        if [ -z "$First_sector" ]; then
                First_sector=2048
        fi
        
        cat ~/Mountlist/Partition.txt | grep Last > ~/Mountlist/Create.txt
        Last_sector=`sed -e 's/Last sector ://g' ~/Mountlist/Create.txt`
        if [ -z "$Last_sector" ]; then
                Last_sector=`expr $First_sector + 10240`
        fi

        cat ~/Mountlist/Partition.txt | grep Select > ~/Mountlist/Create.txt
        Select_Partition_Number=`sed -e 's/Select Partition Number ://g' ~/Mountlist/Create.txt`
        if [ -z "$Select_Partition_Number" ]; then
                Select_Partition_Number=8e
        fi
        

        cat ~/Mountlist/Partition.txt | grep Point > ~/Mountlist/Create.txt
        Mount_Point=`sed -e 's/Mount Point ://g' ~/Mountlist/Create.txt`


        cat ~/Mountlist/Partition.txt | grep Filesystem > ~/Mountlist/Create.txt
        Mount_Filesystem=`sed -e 's/Mount Filesystem ://g' ~/Mountlist/Create.txt`
        if [ -z "$Mount_Filesystem" ]; then
                Mount_Filesystem=ext3
        fi

        echo "살아있음!"
        fdisk $i
        
                for go in n $Partition_type $Create_Partition_Number $First_sector $Last_sector t $Select_Partition_Number w;
                do
                        echo $go

                done

        mkfs.$Mount_Filesystem $i$Num
        mount $Mount_Point

        echo $i$Num     $Mount_Point    ext3    defaults        0       0
        (NUM++) >> /etc/fstab
        done
        rm -rf ~/Mountlist/Mount2.txt ~/Mountlist/Create.txt

}

Main(){
        Mount
}

    #!/bin/bash

Useradd(){
        UserName=`awk '/UserName/' UserTemplates.txt`
        UserName=${UserName#*UserName :}
        if [ -z $UserName ]; then
                echo "User 이름을 확인하세요! "
        fi
        UserName=`echo " $UserName" | tr -d ' '`


        Password=`awk '/Password/' UserTemplates.txt`
        Password=${Password#*Password :}

        UserShell=`awk '/UserShell/' UserTemplates.txt` 
        UserShell=${UserShell#*UserShell :}

        if [ -z $UserShell ]; then
                UserShell='/bin/bash'
        fi

        HomeWhich=`awk '/HomeWhich/' UserTemplates.txt` 
        HomeWhich=${HomeWhich#*HomeWhich :}
        if [ -z $HomeWhich ]; then
                HomeWhich="/home/$UserName"
                HomeWhich=`echo "$HomeWhich" | tr -d ' '`
        fi

        HomeDir=`awk '/HomeDir/' UserTemplates.txt`
        HomeDir=${HomeDir#*HomeDir :}
        if [ -z $HomeDir ]; then
                HomeDir='-m -d '
        elif [ $HomeDir == "-M" ]; then
                HomeWhich=''
                HomeDir=''
        else
                HomeDir='-m -d '
        fi


        UserGroup=`awk '/UserGroup/' UserTemplates.txt` 
        UserGroup=${UserGroup#*UserGroup :}
        if [ -z $UserGroup ]; then
                UserGroup='-N'
        fi

        UserUID=`awk '/UserUID/' UserTemplates.txt` 
        UserUID=${UserUID#*UserUID :}
        if [ ! -z $UserUID ]; then
                UserUID="-u $UserUID"
        fi

        GroupUID=`awk '/GroupUID/' UserTemplates.txt` 
        GroupUID=${GroupUID#*GroupUID :}
        if [ ! -z $GroupUID ]; then
                GroupUID="-g $GroupUID"
        fi



        useradd -n $UserName -s $UserShell $HomeDir $HomeWhich $UserGroup $UserUID $GroupUD

        if [ ! -z $Password ]; then
                echo "$Password" | passwd --stdin "$UserName"
                echo "Password 설정완료"
        fi


}


main(){
Useradd

}

    파일에서 특정 단어를 찾아서 행을 변경하는 스크립트

    #!/bin/bash

# 옵션에 맞게 스크립트를 실행시키세요.

# ex Change.sh [File] [Keyword] [변경할 키워드]

# ex Change.sh /etc/httpd//condf/httpd.conf Listen Listen 81

# 주의할 점은 띄어쓰기를 원할시 반드시 "_"로 표시해주세요.

if [ -z $1 -o -z $2 -o -z $3 ]; then
        echo "Error : 잘못된 옵션을 입력하셨습니다."
else
        Num=`grep -n $2 $1 | sed '/#/d'`
        echo ""
        echo "변경 전 : $Num" 
        echo ""
        Num=${Num%%:*}

        echo $3 > ./Change.txt
        Change=`sed -e 's/_/ /g' Change.txt`
        rm -rf ./Change.txt

        Num=`echo "$Num s/.*/" | tr -d ' '`
        sed -i "$Num$Change/g"  $1
        Num=`grep -n $Change $1 | sed '/#/d'`
        echo ""
        echo "변경 후 : $Num"
        echo ""


fi
